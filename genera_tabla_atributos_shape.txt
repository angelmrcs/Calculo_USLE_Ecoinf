SELECT uc569903_cubillas.OBJECTID, asigna_C_CLC.DESCRIPCION AS des_99, asigna_C_CLC.min_c AS min_c_99, asigna_C_CLC.max_c AS max_c_99, asigna_C_CLC.avg_c AS avg_c_99, asigna_C_CLC_1.DESCRIPCION AS des_03, asigna_C_CLC_1.min_c AS min_c_03, asigna_C_CLC_1.max_c AS max_c_03, asigna_C_CLC_1.avg_c AS avg_c_03, asigna_C_CLC_2.DESCRIPCION AS des_56, asigna_C_CLC_2.min_c AS min_c_56, asigna_C_CLC_2.max_c AS max_c_56, asigna_C_CLC_2.avg_c AS avg_c_56
FROM asigna_C_CLC AS asigna_C_CLC_2 INNER JOIN (asigna_C_CLC AS asigna_C_CLC_1 INNER JOIN (asigna_C_CLC INNER JOIN uc569903_cubillas ON asigna_C_CLC.COD_USO = uc569903_cubillas.UC99) ON asigna_C_CLC_1.COD_USO = uc569903_cubillas.UC03) ON asigna_C_CLC_2.COD_USO = uc569903_cubillas.UC56;
